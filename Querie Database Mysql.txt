CREATE DATABASE PROJETO_ADMISSAO;

CREATE TABLE PARCEIROS
(
	ID_PARCEIROS INT NOT NULL AUTO_INCREMENT,
	NOME						VARCHAR(255),
	TIPO_PARCEIRO				VARCHAR(255),
	TIPO_PESSOA					VARCHAR(255),
	CPF_CNPJ					VARCHAR(255),
	EMAIL						VARCHAR(255),
	STATUS						VARCHAR(255),
	TELEFONE					VARCHAR(255),
	CEP							VARCHAR(255),
	LOGRADOURO					VARCHAR(255),
	NUMERO						VARCHAR(255),
	BAIRRO						VARCHAR(255),
	ESTADO						VARCHAR(255),
	CIDADE						VARCHAR(255),
	
	PRIMARY KEY(ID_PARCEIROS)
);
